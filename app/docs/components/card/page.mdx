## Usage

```tsx
import {
  Card,
  CardContent,
  CardDescription,
  CardHeader,
  CardTitle,
} from "@/components/ui/card"

export function Example() {
  return (
    <Card>
      <CardHeader>
        <CardTitle>Issue 09</CardTitle>
        <CardDescription>Editorial system release notes.</CardDescription>
      </CardHeader>
      <CardContent>Composable content goes here.</CardContent>
    </Card>
  )
}
```
