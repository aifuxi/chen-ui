## Usage

```tsx
import { Avatar, AvatarFallback, AvatarImage } from "@/components/ui/avatar"

export function Example() {
  return (
    <Avatar>
      <AvatarImage src="/avatar.jpg" alt="Avery Stone" />
      <AvatarFallback>AS</AvatarFallback>
    </Avatar>
  )
}
```

## Notes

- Always pair `AvatarImage` with `AvatarFallback`.
- `AvatarRing` is optional and works as a purely decorative wrapper.
